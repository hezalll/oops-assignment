#include <iostream>
using namespace std;

class Student
{
private:
    string name;
    string degree;
    string hostel;
    int roll_no;
    float CGPA;

    void validateCGPA()
    {
        if (CGPA < 0 || CGPA > 10)
        {
            cout << "Invalid CGPA! Setting CGPA to 0.\n";
            CGPA = 0;
        }
    }

public:
    void addDetails()
    {
        cout << "Enter name: ";
        cin >> name;

        cout << "Enter roll no.: ";
        cin >> roll_no;

        cout << "Enter degree: ";
        cin >> degree;

        cout << "Enter hostel: ";
        cin >> hostel;

        cout << "Enter current CGPA: ";
        cin >> CGPA;

        validateCGPA();
    }

    void updateDetails()
    {
        cout << "Enter updated name: ";
        cin >> name;

        cout << "Enter updated roll no.: ";
        cin >> roll_no;

        cout << "Enter updated degree: ";
        cin >> degree;

        cout << "Enter updated hostel: ";
        cin >> hostel;

        cout << "Enter updated CGPA: ";
        cin >> CGPA;

        validateCGPA();
    }

    void updateCGPA()
    {
        cout << "Enter new CGPA: ";
        cin >> CGPA;
        validateCGPA();
    }

    void updateHostel()
    {
        cout << "Enter new hostel: ";
        cin >> hostel;
    }

    void displayDetails()
    {
        cout << "\nName: " << name;
        cout << "\nRoll No.: " << roll_no;
        cout << "\nDegree: " << degree;
        cout << "\nHostel: " << hostel;
        cout << "\nCurrent CGPA: " << CGPA << endl;
    }
};

int main()
{
    int n;
    cout << "Enter number of students: ";
    cin >> n;

    Student s[n];

    for (int i = 0; i < n; i++)
    {
        cout << "\nEnter details of student " << i + 1 << endl;
        s[i].addDetails();
    }

    for (int i = 0; i < n; i++)
    {
        cout << "\nDisplaying details of student " << i + 1 << endl;
        s[i].displayDetails();
    }

    return 0;
}

#include <iostream>
using namespace std;

class Complex {
public:
    float real, imag;

    void input() {
        cout << "Enter real part: ";
        cin >> real;
        cout << "Enter imaginary part: ";
        cin >> imag;
    }

    void sumComplex(Complex c1, Complex c2) {
        real = c1.real + c2.real;
        imag = c1.imag + c2.imag;
    }

    void display() {
        cout << real << " + " << imag << "i" << endl;
    }
};

int main() {
    Complex c1, c2, sum;

    c1.input();
    c2.input();

    sum.sumComplex(c1, c2);

    cout << "Sum = ";
    sum.display();

    return 0;
}
